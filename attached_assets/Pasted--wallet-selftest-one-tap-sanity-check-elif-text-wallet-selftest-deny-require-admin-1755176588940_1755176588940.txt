elif text == "/wallet_balance_usd":
    deny = _require_admin(user)
    if deny: return deny
    try:
        import re, wallets
        uid = user.get("id")
        bal_text = wallets.cmd_wallet_balance(uid)  # must contain "SOL: <number>"

        if not m: return _reply("âš ï¸ Could not parse SOL balance.", status="error")

        from prices import get_sol_price_usd  # stub below

            return _reply(f"{bal_text}\nâ‰ˆ $â€” USD (price unavailable)")
        usd = sol_amt * float(price)
        return _reply(f"{bal_text}\nâ‰ˆ ${usd:,.2f} USD")
    except Exception as e:
        return _reply(f"ğŸ’± Balance (USD) error: {e}", status="error")
